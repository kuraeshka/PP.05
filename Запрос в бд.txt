WITH 
last_material_prices AS (
    SELECT DISTINCT ON (м.код_материала) 
        м.код_материала,
        цм.цена,
        цм.дата
    FROM материалы м
    LEFT JOIN цена_материала цм ON м.код_цены_материала = цм.код_цены_материала
    ORDER BY м.код_материала, цм.дата DESC
),
order_details AS (
    SELECT 
        з.код_заказа,
        сз.код_продукции,
        сз.количество AS кол_во_в_заказе,
        сз.сумма AS позиция_сумма,
        з.дата AS дата_заказа
    FROM заказ з
    JOIN состав_заказа сз ON з.код_заказа = сз.код_заказа
    WHERE з.код_заказа = 1   
),
product_specs AS (
    SELECT 
        od.код_заказа,
        od.кол_во_в_заказе,
        od.позиция_сумма,
        od.дата_заказа,
        п.код_спецификации,
        с.количество AS кол_во_по_спецификации,  
        сс.код_материала,
        сс.количество AS норма_материала         
    FROM order_details od
    JOIN продукция п ON od.код_продукции = п.код_продукции
    JOIN спецификация с ON п.код_спецификации = с.код_спецификации
    JOIN состав_спецификации сс ON с.код_спецификации = сс.код_спецификации
),
material_needs AS (
    SELECT
        код_заказа,
        код_материала,
        SUM(кол_во_в_заказе * норма_материала / кол_во_по_спецификации) AS потребность
    FROM product_specs
    GROUP BY код_заказа, код_материала
),
material_costs AS (
    SELECT
        мн.код_заказа,
        мн.код_материала,
        мн.потребность,
        мн.потребность * COALESCE(lmp.цена, 0) AS стоимость_материала
    FROM material_needs мн
    LEFT JOIN last_material_prices lmp ON мн.код_материала = lmp.код_материала
)
SELECT 
    з.код_заказа,
    з.дата,
    кп.имя AS покупатель,
    кпр.имя AS продавец,
    SUM(сз.сумма) AS общая_сумма_заказа,
    COALESCE(SUM(mc.стоимость_материала), 0) AS общая_стоимость_материалов
FROM заказ з
LEFT JOIN состав_заказа сз ON з.код_заказа = сз.код_заказа
LEFT JOIN контрагента кп ON з.код_покупателя = кп.код_контрагента
LEFT JOIN контрагента кпр ON з.код_продавца = кпр.код_контрагента
LEFT JOIN material_costs mc ON з.код_заказа = mc.код_заказа
WHERE з.код_заказа = 1   -- можно убрать или изменить
GROUP BY з.код_заказа, з.дата, кп.имя, кпр.имя;
